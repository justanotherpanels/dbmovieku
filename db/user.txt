-- SQL Schema for Users table in Supabase/PostgreSQL

-- Create custom types for enum-like behavior
CREATE TYPE user_level AS ENUM ('Member', 'Admin');
CREATE TYPE user_status AS ENUM ('Active', 'Not-Active');

CREATE TABLE public.users (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  name TEXT NOT NULL,
  email TEXT UNIQUE NOT NULL,
  password TEXT NOT NULL, -- Pastikan untuk menyimpan password yang sudah di-hash!
  level user_level DEFAULT 'Member' NOT NULL,
  apikey TEXT UNIQUE DEFAULT encode(gen_random_bytes(24), 'base64'),
  status user_status DEFAULT 'Active' NOT NULL,
  create_at TIMESTAMPTZ DEFAULT NOW() NOT NULL,
  update_at TIMESTAMPTZ DEFAULT NOW() NOT NULL
);

-- Trigger to automatically update update_at on changes
CREATE OR REPLACE FUNCTION update_updated_at_column()
RETURNS TRIGGER AS $$
BEGIN
    NEW.update_at = NOW();
    RETURN NEW;
END;
$$ language 'plpgsql';

CREATE TRIGGER update_users_updated_at
    BEFORE UPDATE ON public.users
    FOR EACH ROW
    EXECUTE PROCEDURE update_updated_at_column();
