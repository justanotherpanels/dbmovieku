-- Membuat tabel baru server_2 untuk MovieDB Engine
CREATE TABLE public.server_2 (
    id UUID DEFAULT gen_random_uuid() PRIMARY KEY,
    folder_fld_id TEXT UNIQUE DEFAULT NULL,
    folder_total_files INTEGER DEFAULT NULL,
    folder_name TEXT DEFAULT NULL,
    folder_code TEXT DEFAULT NULL,
    file_splash_img TEXT,
    file_fld_id TEXT,
    file_code TEXT UNIQUE,
    file_canplay INTEGER DEFAULT 0,
    file_download_url TEXT,
    file_length INTEGER,
    file_public INTEGER DEFAULT 1,
    file_uploaded TIMESTAMPTZ,
    file_single_img TEXT,
    title TEXT,
    views INTEGER DEFAULT 0,
    view_count INTEGER DEFAULT 0,
    create_at TIMESTAMPTZ DEFAULT now(),
    update_at TIMESTAMPTZ DEFAULT now()
);

-- Mengaktifkan Row Level Security (RLS)
ALTER TABLE public.server_2 ENABLE ROW LEVEL SECURITY;

-- Memberikan akses read untuk semua orang (anon/authenticated)
CREATE POLICY "Izinkan baca untuk publik" ON public.server_2 FOR SELECT USING (true);
