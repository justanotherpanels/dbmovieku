-- SQL Schema for setting_site table in Supabase/PostgreSQL

CREATE TABLE public.setting_site (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  name TEXT NOT NULL,
  favicon TEXT NULL, -- URL ke file favicon
  logo TEXT NULL, -- URL ke file logo
  create_at TIMESTAMPTZ DEFAULT NOW() NOT NULL,
  update_at TIMESTAMPTZ DEFAULT NOW() NOT NULL
);

-- Trigger to automatically update update_at on changes
CREATE TRIGGER update_setting_site_updated_at
    BEFORE UPDATE ON public.setting_site
    FOR EACH ROW
    EXECUTE PROCEDURE update_updated_at_column();
