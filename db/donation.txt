-- SQL Schema for Donation table in Supabase/PostgreSQL

-- Create custom type for donation methods
CREATE TYPE donation_type AS ENUM ('Qris', 'Crypto', 'Paypal');

CREATE TABLE public.donation (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  type donation_type NOT NULL,
  detail_donation JSONB DEFAULT '{}'::jsonb NOT NULL,
  create_at TIMESTAMPTZ DEFAULT NOW() NOT NULL,
  update_at TIMESTAMPTZ DEFAULT NOW() NOT NULL
);

-- Trigger to automatically update update_at on changes
CREATE OR REPLACE FUNCTION update_donation_updated_at_column()
RETURNS TRIGGER AS $$
BEGIN
    NEW.update_at = NOW();
    RETURN NEW;
END;
$$ language 'plpgsql';

CREATE TRIGGER update_donation_updated_at
    BEFORE UPDATE ON public.donation
    FOR EACH ROW
    EXECUTE PROCEDURE update_donation_updated_at_column();
